### 문제 풀이 과정

```java
import sys

input = sys.stdin.readline

N = int(input())
A = list(map(int, input().split()))

Operator = list(map(int, input().split()))
Operator_sum = 0
for i in range(4):
    Operator_sum += Operator[i]

Operator_possible = [0]*Operator_sum
visited = [False]*Operator_sum

Oper = [0] * Operator_sum
x = 0
for i in range(4):
    for j in range(Operator[i]):
        Oper[x] = i+1
        x += 1

num = []

def Perm(cnt, Operator_sum, Oper, A, N):
    if cnt == Operator_sum:
        # print(Operator_possible)
        ans = A[0]
        for i in range(1, N):
            if Operator_possible[i-1] == 1:
                ans += A[i]
            elif Operator_possible[i-1] == 2:
                ans -= A[i]
            elif Operator_possible[i-1] == 3:
                ans *= A[i]
            elif Operator_possible[i-1] == 4:
                sign = -1 if ans < 0 else 1 
                ans = abs(ans)//A[i]
                ans *= sign
        num.append(ans)
        return
    
    for i in range(Operator_sum):
        if(visited[i]):
            continue

        Operator_possible[cnt] = Oper[i]
        visited[i] = True;
        Perm(cnt+1, Operator_sum, Oper, A, N)
        visited[i] = False;



Perm(0, Operator_sum, Oper, A, N)
print(max(num))
print(min(num))
```
